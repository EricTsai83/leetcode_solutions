class Solution:
    def isValid(self, s: str) -> bool:
        # edge case
        if len(s) <= 1:
            return False
        
        # 左括號塞進去，右括號pop出來判斷
        # 寫try是如果碰到一個右括號要pop但是tmp_stack是空的，代表直接錯誤了可以return
        tmp_stack = []
        for i, elem in enumerate(s):
            try:
                if elem in set(['{', '(', '[']):
                    tmp_stack.append(elem)
                if elem == '}':
                    if tmp_stack.pop() != '{':
                        return False
                if elem == ')':
                    if tmp_stack.pop() != '(':
                        return False
                if elem == ']':
                    if tmp_stack.pop() != '[':
                        return False
            except:
                return False
        
        # 全部跑完後，stack容器要是空的 -> 不然代表有問題
        if len(tmp_stack) == 0:
            return True
        else: 
            return False
